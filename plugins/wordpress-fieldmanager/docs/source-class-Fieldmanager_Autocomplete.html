 91     /**
 92      * Render form element
 93      * @param mixed $value
 94      * @return string HTML
 95      */
 96     public function form_element( $value = Null ) {
 97 
 98         if ( $this->exact_match ) {
 99             $this->attributes['data-exact-match'] = True;
100         }
101 
102         if ( $this->datasource->use_ajax ) {
103             $this->attributes['data-action'] = $this->datasource->get_ajax_action( $this->name );
104             list ( $context, $subcontext ) = fm_get_context();
105             $this->attributes['data-context'] = $context;
106             $this->attributes['data-subcontext'] = $subcontext;
107         } else {
108             $this->attributes['data-options'] = htmlspecialchars( json_encode( $this->datasource->get_items() ) );
109         }
110 
111         $element = sprintf(
112             '<input class="fm-autocomplete fm-element fm-incrementable" type="text" id="%s" value="%s" %s />',
113             $this->get_element_id(),
114             $this->datasource->get_value( $value ),
115             $this->get_element_attributes()
116         );
117 
118         $element .= sprintf(
119             '<input class="fm-autocomplete-hidden fm-element" type="hidden" name="%s" value="%s" />',
120             $this->get_form_name(),
121             $value
122         );
123 
124         if ( isset( $this->show_view_link ) && $this->show_view_link ) {
125             $element .= $this->datasource->get_view_link( $value );
126         }
127 
128         if ( isset( $this->show_edit_link ) && $this->show_edit_link ) {
129             $element .= $this->datasource->get_edit_link( $value );
130         }
131 
132         return $element;
133     }

155     /**
156      * Helper function to get the list of default meta boxes to remove.
157      * If $remove_default_meta_boxes is true and the datasource is Fieldmanager_Datasource_Term,
158      * this will return a list of all default meta boxes for the specified taxonomies.
159      * We only need to return id and context since the page will be handled by the list of post types provided to add_meta_box.
160      * Otherwise, this will just return an empty array.
161      * @param array current list of meta boxes to remove
162      * @return array list of meta boxes to remove
163      */
164     protected function add_meta_boxes_to_remove( &$meta_boxes_to_remove ) {
165         if ( $this->remove_default_meta_boxes && get_class( $this->datasource ) == 'Fieldmanager_Datasource_Term' ) {
166             // Iterate over the list and build the list of meta boxes
167             $meta_boxes = array();
168             foreach( $this->datasource->get_taxonomies() as $taxonomy ) {
169                 // The ID differs if this is a hierarchical taxonomy or not. Get the taxonomy object.
170                 $taxonomy_obj = get_taxonomy( $taxonomy );
171                 if ( false !== $taxonomy_obj ) {
172                     if ( $taxonomy_obj->hierarchical )
173                         $id = $taxonomy . "div";
174                     else
175                         $id = 'tagsdiv-' . $taxonomy;
176 
177                     $meta_boxes[$id] = array(
178                         'id' => $id,
179                         'context' => 'side'
180                     );
181                 }
182             }
183 
184             // Merge in the new meta boxes to remove
185             $meta_boxes_to_remove = array_merge( $meta_boxes_to_remove, $meta_boxes );
186         }
187     }
