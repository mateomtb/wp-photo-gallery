 53     /**
 54      * Add a context to a fieldmanager
 55      * @param string|string[] $taxonomies
 56      * @param boolean $show_on_add Whether or not to show the fields on the add term form
 57      * @param boolean $show_on_edit Whether or not to show the fields on the edit term form
 58      * @param Fieldmanager_Field $fm
 59      */
 60     public function __construct( $title, $taxonomies, $show_on_add = true, $show_on_edit = true, $parent = '', $fm = null ) {
 61         // Populate the list of taxonomies for which to add this meta box with the given settings
 62         if ( !is_array( $taxonomies ) ) $taxonomies = array( $taxonomies );
 63 
 64         // Set the class variables
 65         $this->title = $title;
 66         $this->taxonomies = $taxonomies;
 67         $this->show_on_add = $show_on_add;
 68         $this->show_on_edit = $show_on_edit;
 69         $this->parent = $parent;
 70         $this->fm = $fm;
 71 
 72         // Iterate through the taxonomies and add the fields to the requested forms
 73         // Also add handlers for saving the fields and which forms to validate (if enabled)
 74         foreach ( $taxonomies as $taxonomy ) {
 75             if ( $this->show_on_add ) {
 76                 add_action( $taxonomy . '_add_form_fields', array( $this, 'add_term_fields' ), 10, 1 );
 77                 add_action( 'created_term', array( $this, 'save_term_fields'), 10, 3 );
 78             }
 79 
 80             if ( $this->show_on_edit ) {
 81                 add_action( $taxonomy . '_edit_form_fields', array( $this, 'edit_term_fields' ), 10, 2 );
 82                 add_action( 'edited_term', array( $this, 'save_term_fields'), 10, 3 );
 83             }
 84 
 85             // Also handle removing data when a term is deleted
 86             add_action( 'delete_term', array( $this, 'delete_term_fields'), 10, 4 );
 87         }

138     /**
139      * Generates HTML for the term field
140      * @access public
141      * @param string $html_template THe HTML used to wrap the field
142      * @param string $taxonomy
143      * @param WP_Term $term
144      * @return string The element markup
145      */
146     public function term_fields( $html_template, $taxonomy, $term = null ) {
147         // Make sure the user hasn't specified a field name we can't use
148         if ( in_array( $this->fm->name, $this->reserved_fields ) )
149             $this->fm->_invalid_definition( 'The field name ' . $this->fm->name . ' is reserved for WordPress on the term form.' );
150 
151         // Check if there are any current values to retrieve
152         if ( isset( $term->term_id ) ) {
153             $term_meta = Fieldmanager_Util_Term_Meta();
154             $values = $term_meta->get_term_meta( $term->term_id, $taxonomy, $this->fm->name, true );
155         }
156         $values = empty( $values ) ? null : $values;
157 
158         // Set the data type and ID
159         $this->fm->data_type = 'term';
160         $this->fm->data_id = is_object( $term ) ? $term->term_id : null;
161 
162         // Create the display label if one is set
163         if ( ! empty( $this->title ) ) {
164             $label = sprintf(
165                 '<label for="%s">%s</label>',
166                 esc_attr( $this->fm->name ),
167                 esc_html( $this->title )
168             );
169         } else {
170             $label = '';
171         }
172 
173         // Create the markup and return it
174         return sprintf(
175             $html_template,
176             $label,
177             wp_nonce_field( 'fieldmanager-save-' . $this->fm->name, 'fieldmanager-' . $this->fm->name . '-nonce', true, false ),
178             $this->fm->element_markup( $values )
179         );
180     }

182     /**
183      * Saves custom term fields
184      * @access public
185      * @param int $term_id
186      * @param int $tt_id
187      * @param string $taxonomy
188      * @return void
189      */
190     public function save_term_fields( $term_id, $tt_id, $taxonomy ) {
191         // Make sure this field is attached to the taxonomy being saved and this is the appropriate action
192         if ( ! in_array( $taxonomy, $this->taxonomies ) )
193             return;
194 
195         // Make sure the current user can save this post
196         $tax_obj = get_taxonomy( $taxonomy );
197         if( !current_user_can( $tax_obj->cap->manage_terms ) ) {
198             $this->fm->_unauthorized_access( 'User cannot edit this term' );
199             return;
200         }
201 
202         // Make sure that our nonce field arrived intact
203         if( ! wp_verify_nonce( $_POST['fieldmanager-' . $this->fm->name . '-nonce'], 'fieldmanager-save-' . $this->fm->name ) ) {
204             $this->fm->_unauthorized_access( 'Nonce validation failed' );
205         }
206 
207         // Save the data
208         $value = isset( $_POST[ $this->fm->name ] ) ? $_POST[ $this->fm->name ] : "";
209         $this->save_to_term_meta( $term_id, $taxonomy, $value );
210     }

212     /**
213      * Helper to save an array of data to term meta
214      * @param int $term_id
215      * @param array $data
216      * @return void
217      */
218     public function save_to_term_meta( $term_id, $taxonomy, $data ) {
219         // Set the data ID and type
220         $this->fm->data_id = $term_id;
221         $this->fm->data_type = 'term';
222 
223         // Get an instance of the term meta class
224         $term_meta = Fieldmanager_Util_Term_Meta();
225 
226         // Get the current data and run presave
227         $current = $term_meta->get_term_meta( $this->fm->data_id, $taxonomy, $this->fm->name, true );
228         $data = $this->fm->presave_all( $data, $current );
229 
230         // Unless we are skipping the save, store the new value in term meta
231         if ( ! $this->fm->skip_save )
232             $term_meta->update_term_meta( $term_id, $taxonomy, $this->fm->name, $data );
233     }
