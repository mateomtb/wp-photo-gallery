 92      * Form element

 95      */
 96     public function form_element( $value = array() ) {
 97         if ( is_numeric( $value ) && $value > 0 ) {
 98             $attachment = get_post( $value );
 99             if ( strpos( $attachment->post_mime_type, 'image/' ) === 0 ) {
100                 $preview = sprintf( '%s<br />', __( 'Uploaded image:' ) );
101                 $preview .= '<a href="#">' . wp_get_attachment_image( $value, $this->preview_size, false, array( 'class' => $this->thumbnail_class ) ) . '</a>';
102             } else {
103                 $preview = sprintf( '%s', __( 'Uploaded file:' ) ) . '&nbsp;';
104                 $preview .= wp_get_attachment_link( $value, $this->preview_size, True, True, $attachment->post_title );
105             }
106             $preview .= sprintf( '<br /><a href="#" class="fm-media-remove fm-delete">%s</a>', __( 'remove' ) );
107             $preview = apply_filters( 'fieldmanager_media_preview', $preview, $value, $attachment );
108         } else {
109             $preview = '';
110         }
111         return sprintf(

114             <div class="media-wrapper">%5$s</div>
115             <script type="text/javascript">
116             var fm_preview_size = fm_preview_size || [];
117             fm_preview_size["%1$s"]=%6$s;
118             </script>',
119             $this->get_element_id(),
120             $this->get_form_name(),
121             esc_attr( $this->button_label ),
122             esc_attr( $value ),
